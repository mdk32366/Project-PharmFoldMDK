# Runtime dependencies — the Fly serving tier (D-013).
#
# EXACT pins only. A range means the gate's behaviour can change with no commit in
# this repo, producing a red `main` with an empty git log to explain it. Upgrades
# happen in a PR, where the gate proves them.
#
# NOT here, deliberately: torch, transformers, bitsandbytes, ESMFold weights.
# Those belong to the local/rented GPU tier (D-004, D-011) and get their own
# manifest when `worker/` is built. CI runs on a CPU runner and must never try to
# install a CUDA stack.

SQLAlchemy==2.0.51
alembic==1.18.5

# The Fly serving tier (D-031): FastAPI app + the four worker→Fly routes. `uvicorn`
# is the ASGI server prod runs the app under; `python-multipart` is REQUIRED for
# FastAPI to parse the `/artifacts` multipart upload (pdb / plddt / pae / provenance).
# Plain `uvicorn` (not [standard]): the uvloop/httptools extras have no universal
# wheel set and Fly's linux runtime does not need them for single-worker scale (D-004).
fastapi==0.121.2
uvicorn==0.38.0
python-multipart==0.0.20

# Postgres driver for prod (D-012). psycopg 3, not psycopg2 — actively developed
# and the current SQLAlchemy 2.0 recommendation. [binary] ships a wheel with libpq
# bundled, so no system headers are needed at install time.
psycopg[binary]==3.3.4
